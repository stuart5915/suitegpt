2

Add the widget script

‚ñº

Copy this code into Scriptable. Your Telegram ID is already filled in!

SUITE-Widget.js 
// SUITE Widget - Tasks & Groceries
const TELEGRAM_ID = "YOUR_ID";
const API = "https://rdsmdywbdiskxknluiym.supabase.co/functions/v1/widget-get-lists";

async function getLists() {
  const req = new Request(API);
  req.method = "POST";
  req.headers = {"Content-Type": "application/json"};
  req.body = JSON.stringify({telegram_id: TELEGRAM_ID});
  return (await req.loadJSON()).lists || [];
}

const w = new ListWidget();
w.backgroundGradient = (() => {
  const g = new LinearGradient();
  g.locations = [0, 1];
  g.colors = [new Color("#1a1a2e"), new Color("#16213e")];
  return g;
})();
w.setPadding(14, 14, 14, 14);

const title = w.addText("SUITE");
title.font = Font.boldSystemFont(13);
title.textColor = new Color("#ff9500");
w.addSpacer(6);

const lists = await getLists();
for (const list of lists.slice(0, 2)) {
  const h = w.addStack();
  const icon = h.addText(list.icon + " " + list.name);
  icon.font = Font.mediumSystemFont(12);
  icon.textColor = Color.white();
  h.addSpacer();
  const count = h.addText(String(list.incomplete_count || 0));
  count.font = Font.boldSystemFont(12);
  count.textColor = new Color("#ff9500");
  w.addSpacer(4);

  for (const item of (list.items||[]).filter(i=>!i.is_completed).slice(0,3)) {
    const t = w.addText("  ‚Ä¢ " + item.content);
    t.font = Font.systemFont(11);
    t.textColor = new Color("#ffffff", 0.8);
  }
  w.addSpacer(8);
}

w.url = "https://getsuite.app/lists";
Script.setWidget(w);
Script.complete();
