Download Scriptable

Add the Widget Script

Copy this script and paste it into a new Scriptable script. Run it once to configure your Telegram ID. 

// SUITE Widget - paste into Scriptable
// After pasting, run the script once to set up

const SUPABASE_URL = "https://rdsmdywbdiskxknluiym.supabase.co";
const API = SUPABASE_URL + "/functions/v1/widget-get-lists";

// Your Telegram ID - replace with your actual ID
const TELEGRAM_ID = "YOUR_TELEGRAM_ID_HERE";

async function fetchLists() {
  const req = new Request(API);
  req.method = "POST";
  req.headers = { "Content-Type": "application/json" };
  req.body = JSON.stringify({ telegram_id: TELEGRAM_ID });
  return (await req.loadJSON()).lists || [];
}

const widget = new ListWidget();
const gradient = new LinearGradient();
gradient.locations = [0, 1];
gradient.colors = [new Color("#1a1a2e"), new Color("#16213e")];
widget.backgroundGradient = gradient;
widget.setPadding(16, 16, 16, 16);

const header = widget.addText("SUITE");
header.font = Font.boldSystemFont(14);
header.textColor = new Color("#ff9500");
widget.addSpacer(8);

const lists = await fetchLists();
for (const list of lists.slice(0, 2)) {
  const row = widget.addStack();
  row.addText(list.icon + " " + list.name);
  row.addSpacer();
  row.addText(list.incomplete_count + "");
  widget.addSpacer(4);

  for (const item of list.items.filter(i => !i.is_completed).slice(0, 3)) {
    const t = widget.addText("  " + item.content);
    t.font = Font.systemFont(11);
    t.textColor = Color.white();
  }
  widget.addSpacer(8);
}

widget.url = "https://getsuite.app/lists";
widget.refreshAfterDate = new Date(Date.now() + 15*60*1000);
Script.setWidget(widget);
Script.complete();
  1. 1 Open Scriptable and tap + to create a new script
  2. 2 Paste the code above
  3. 3 Replace YOUR_TELEGRAM_ID_HERE with your actual Telegram ID
  4. 4 Tap the play button to run and test

Add Widget to Home Screen

  1. 1 Long-press on your home screen and tap the + button
  2. 2 Search for Scriptable
  3. 3 Choose your preferred widget size (Medium recommended)
  4. 4 Long-press the widget and tap Edit Widget
  5. 5 Select your SUITE Widget script

  2. 2 Create a new shortcut with these actions:

  3. 3 Name it "Add to SUITE" and add to home screen
